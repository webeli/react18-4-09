import React from 'react';

export class DoubleClicker extends React.Component {
  constructor(props){
    super(props);
    this.state = {counter: 0};
    this.increaseTwice = this.increaseTwice.bind(this);
  }
  increaseTwice() {
    this.setState(state => ({counter: state.counter + 1}));
    this.setState(state => ({counter: state.counter + 1}));
  }
  render() {
    return (
      <div>
        <p>This will increase twice! :)</p>
        <button onClick={this.increaseTwice}>
          Count is {this.state.counter}
        </button>
      </div>
    );
  }
}
