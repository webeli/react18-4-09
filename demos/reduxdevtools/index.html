import {bindActionCreators} from 'redux';

import {store} from './store';

export const actionCreators = {
  increment(int) {
    return {type: 'INCREMENT', by: int}
  }
};

export const boundActionCreators = bindActionCreators(
  actionCreators,
  store.dispatch
);

export const deaf = middlewareAPI => {
  let i = 0;
  return next => action => {
    if (!(i++%3)) {
      next(action);
    }
  };
};

export const nervous = middlewareAPI => next => action => {
  next(action);
  next(action);
};

export const impatient = middlewareAPI => next => action => {
  action = Object.assign({},action); // copy
  action.by *= 5;
  next(action);
};

export const logger = middlewareAPI => next => action => {
  console.log("Dispatching action:",action);
  let result = next(action);
  let newstate = middlewareAPI.getState();
  console.log("State after dispatch:",newstate);
  return result;
};

import { createStore } from 'redux';

let initialstate = {count: 0};

let reducer = (state,action)=> {
  switch(action.type){
    case 'INCREMENT': return {count: state.count + action.by};
    default: return state;
  }
};

export const store = createStore(
  reducer,
  initialstate,
  window.devToolsExtension && window.devToolsExtension()
);
