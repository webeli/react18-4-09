import React from 'react';

import Perimeter from 'react-perimeter';

export class Screen extends React.Component {
  constructor(props){
    super(props);
    this.state = {alarm: false};
  }
  render(){
    let alarmStyle = {backgroundColor: 'red', color: 'white'};

    return (
      <div style={this.state.alarm ? alarmStyle : {}}>
        <p>Lorem ipsum</p>
        <p>Dolor sit amet</p>
        <Perimeter onBreach={()=>this.setState({alarm:true})} padding={80}>
          <button>load more</button>
        </Perimeter>
      </div>
    );
  }
}
