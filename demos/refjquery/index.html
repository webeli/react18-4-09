import React from 'react';
import ReactDOM from 'react-dom';

import {Text} from './text';

import './swedify'; // register our plugin in jQuery

ReactDOM.render(
	<Text/>,
	document.getElementById("app")
);

import $ from 'jquery';

$.fn.swedify = function(){
	return $(this).css({
		border: "5px solid blue",
		backgroundColor: "yellow",
		padding: 5
	});
}
import React from 'react';
import $ from 'jquery';

export const Text = ()=> (
	<div>
		<p ref={(el) => $(el).swedify()}>
			I will be Swedified!
		</p>
	</div>
);
