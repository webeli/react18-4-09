import React from 'react';
import {string} from 'prop-types';

export class Provider extends React.Component {
  static get childContextTypes() {
    return {
      title: string
    };
  }
  getChildContext() {
    return {
      title: this.props.headline
    };
  }
  render(){
    return this.props.children;
  }
};
