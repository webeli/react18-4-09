import React from 'react';

export class Form extends React.Component {
	constructor(props){
		super(props);
		this.state = {
			submittedname: ''
		};
		this.submitName = this.submitName.bind(this);
	}
	submitName() {
		this.setState({submittedname: this.field.value});
	}
	render() {
		let current = this.state.submittedname;
		return <div>
			<fieldset>
				<legend>Enter your name</legend>
				<input type="text" ref={el=>this.field=el} defaultValue={current} />
				<button onClick={this.submitName}>Submit!</button>
			</fieldset>
			{current ? 'Is '+current+' really your real name?' : 'Who are you?'}
		</div>;
	}
}
